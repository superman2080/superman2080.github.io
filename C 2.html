/*
	C언어 모래시계
*/
include 

int main()
{
	int cnt = 0;
	printf("세로 길이를 입력하세요. :");
	scanf_s("%d", &cnt);

	for (int i = 0; i <= cnt / 2 - 1; i++)
	{
		for (int j = 0; j < i; j++)
		{
			printf(" ");
		}
		for (int j = cnt - i * 2 - 1; j >= 0; j--)
		{
			printf("*");
		}
		printf("\n");
	}
	for (int i = 1; i <= cnt / 2 + 1; i++)
	{
		for (int j = cnt - i - cnt / 2 - 1; j >= 0; j--)
		{
			printf(" ");
		}
		for (int j = 0; j < i * 2 - 1; j++)
		{
			printf("*");
		}
		printf("\n");
	}

	return 0;
}
